#### 第二步：创建 API 方法

```typescript
// plus-ui/src/api/demo/demo/index.ts

import request from '@/utils/request';
import { DemoVO, DemoForm, DemoQuery } from './types';

export const listDemo = (query?: DemoQuery) => {
  return request({
    url: '/demo/demo/list',
    method: 'get',
    params: query
  });
};

export const getDemo = (id: string | number) => {
  return request({
    url: '/demo/demo/' + id,
    method: 'get'
  });
};

export const addDemo = (data: DemoForm) => {
  return request({
    url: '/demo/demo',
    method: 'post',
    data: data
  });
};

export const updateDemo = (data: DemoForm) => {
  return request({
    url: '/demo/demo',
    method: 'put',
    data: data
  });
};

export const delDemo = (id: string | number | Array<string | number>) => {
  return request({
    url: '/demo/demo/' + id,
    method: 'delete'
  });
};
```

#### 第三步：创建 Vue 组件

```vue
<!-- plus-ui/src/views/demo/demo/index.vue -->

<template>
  <div class="p-2">
    <!-- 搜索表单 -->
    <el-card shadow="hover">
      <el-form ref="queryFormRef" :model="queryParams" :inline="true">
        <el-form-item label="key键" prop="testKey">
          <el-input v-model="queryParams.testKey" placeholder="请输入key键" 
                    clearable @keyup.enter="handleQuery" />
        </el-form-item>
        <el-form-item>
          <el-button type="primary" icon="Search" @click="handleQuery">搜索</el-button>
          <el-button icon="Refresh" @click="resetQuery">重置</el-button>
        </el-form-item>
      </el-form>
    </el-card>

    <!-- 数据表格 -->
    <el-card shadow="hover" class="mt-3">
      <template #header>
        <el-row :gutter="10">
          <el-col :span="1.5">
            <el-button v-hasPermi="['demo:demo:add']" type="primary" plain icon="Plus" 
                       @click="handleAdd">新增</el-button>
          </el-col>
          <el-col :span="1.5">
            <el-button v-hasPermi="['demo:demo:edit']" type="success" plain icon="Edit" 
                       :disabled="single" @click="handleUpdate()">修改</el-button>
          </el-col>
          <el-col :span="1.5">
            <el-button v-hasPermi="['demo:demo:remove']" type="danger" plain icon="Delete" 
                       :disabled="multiple" @click="handleDelete()">删除</el-button>
          </el-col>
        </el-row>
      </template>

      <el-table v-loading="loading" border :data="demoList" @selection-change="handleSelectionChange">
        <el-table-column type="selection" width="55" align="center" />
        <el-table-column label="主键" align="center" prop="id" />
        <el-table-column label="key键" align="center" prop="testKey" />
        <el-table-column label="值" align="center" prop="value" />
        <el-table-column label="操作" align="center" width="100">
          <template #default="scope">
            <el-button v-hasPermi="['demo:demo:edit']" link type="primary" icon="Edit" 
                       @click="handleUpdate(scope.row)"></el-button>
            <el-button v-hasPermi="['demo:demo:remove']" link type="primary" icon="Delete" 
                       @click="handleDelete(scope.row)"></el-button>
          </template>
        </el-table-column>
      </el-table>

      <pagination v-show="total > 0" v-model:page="queryParams.pageNum" 
                  v-model:limit="queryParams.pageSize" :total="total" @pagination="getList" />
    </el-card>

    <!-- 添加或修改对话框 -->
    <el-dialog v-model="dialog.visible" :title="dialog.title" width="500px">
      <el-form ref="demoFormRef" :model="form" :rules="rules" label-width="80px">
        <el-form-item label="key键" prop="testKey">
          <el-input v-model="form.testKey" placeholder="请输入key键" />
        </el-form-item>
        <el-form-item label="值" prop="value">
          <el-input v-model="form.value" placeholder="请输入值" />
        </el-form-item>
      </el-form>
      <template #footer>
        <el-button :loading="buttonLoading" type="primary" @click="submitForm">确 定</el-button>
        <el-button @click="cancel">取 消</el-button>
      </template>
    </el-dialog>
  </div>
</template>

<script setup name="Demo" lang="ts">
import { listDemo, getDemo, delDemo, addDemo, updateDemo } from '@/api/demo/demo';
import { DemoVO, DemoQuery, DemoForm } from '@/api/demo/demo/types';

const { proxy } = getCurrentInstance() as ComponentInternalInstance;

const demoList = ref<DemoVO[]>([]);
const loading = ref(true);
const buttonLoading = ref(false);
const ids = ref<Array<string | number>>([]);
const single = ref(true);
const multiple = ref(true);
const total = ref(0);

const queryFormRef = ref<ElFormInstance>();
const demoFormRef = ref<ElFormInstance>();

const dialog = reactive<DialogOption>({
  visible: false,
  title: ''
});

const initFormData: DemoForm = {
  id: undefined,
  testKey: undefined,
  value: undefined
};

const data = reactive<PageData<DemoForm, DemoQuery>>({
  form: { ...initFormData },
  queryParams: {
    pageNum: 1,
    pageSize: 10,
    testKey: undefined,
    value: undefined
  },
  rules: {
    testKey: [{ required: true, message: 'key键不能为空', trigger: 'blur' }],
    value: [{ required: true, message: '值不能为空', trigger: 'blur' }]
  }
});

const { queryParams, form, rules } = toRefs(data);

const getList = async () => {
  loading.value = true;
  const res = await listDemo(queryParams.value);
  demoList.value = res.rows;
  total.value = res.total;
  loading.value = false;
};

const handleQuery = () => {
  queryParams.value.pageNum = 1;
  getList();
};

const resetQuery = () => {
  queryFormRef.value?.resetFields();
  handleQuery();
};

const handleSelectionChange = (selection: DemoVO[]) => {
  ids.value = selection.map((item) => item.id);
  single.value = selection.length != 1;
  multiple.value = !selection.length;
};

const handleAdd = () => {
  form.value = { ...initFormData };
  dialog.visible = true;
  dialog.title = '添加';
};

const handleUpdate = async (row?: DemoVO) => {
  form.value = { ...initFormData };
  const _id = row?.id || ids.value[0];
  const res = await getDemo(_id);
  Object.assign(form.value, res.data);
  dialog.visible = true;
  dialog.title = '修改';
};

const submitForm = () => {
  demoFormRef.value?.validate(async (valid: boolean) => {
    if (valid) {
      buttonLoading.value = true;
      if (form.value.id) {
        await updateDemo(form.value).finally(() => (buttonLoading.value = false));
      } else {
        await addDemo(form.value).finally(() => (buttonLoading.value = false));
      }
      proxy?.$modal.msgSuccess('操作成功');
      dialog.visible = false;
      await getList();
    }
  });
};

const handleDelete = async (row?: DemoVO) => {
  const _ids = row?.id || ids.value;
  await proxy?.$modal.confirm('是否确认删除？').finally(() => (loading.value = false));
  await delDemo(_ids);
  proxy?.$modal.msgSuccess('删除成功');
  await getList();
};

const cancel = () => {
  form.value = { ...initFormData };
  dialog.visible = false;
};

onMounted(() => {
  getList();
});
</script>
```

## 三、常见问题解决

### 3.1 权限问题

**问题**: 用户无法访问某个功能
**解决**:
1. 检查用户是否有对应权限
2. 检查权限标识是否正确
3. 检查 Controller 上的 `@SaCheckPermission` 注解

### 3.2 数据权限问题

**问题**: 用户看不到某些数据
**解决**:
1. 检查 Mapper 上的 `@DataPermission` 注解
2. 检查数据权限配置
3. 检查用户的部门和角色

### 3.3 前端请求失败

**问题**: 前端调用 API 返回 404 或 500
**解决**:
1. 检查 API 路径是否正确
2. 检查后端 Controller 的 `@RequestMapping` 路径
3. 查看后端日志获取详细错误信息

### 3.4 表单验证失败

**问题**: 表单无法提交
**解决**:
1. 检查表单规则是否正确
2. 检查字段是否填写
3. 检查字段类型是否匹配

## 四、性能优化建议

1. **使用分页查询**: 避免一次性加载大量数据
2. **使用缓存**: 对频繁查询的数据进行缓存
3. **使用索引**: 为常用查询字段添加数据库索引
4. **使用异步操作**: 对耗时操作使用异步处理
5. **使用 CDN**: 对静态资源使用 CDN 加速
6. **使用压缩**: 对响应数据进行 gzip 压缩

## 五、安全建议

1. **参数校验**: 对所有输入参数进行校验
2. **权限控制**: 使用权限注解进行权限控制
3. **数据加密**: 对敏感数据进行加密存储
4. **SQL 注入防护**: 使用参数化查询
5. **XSS 防护**: 对用户输入进行转义
6. **CSRF 防护**: 使用 Token 进行 CSRF 防护
